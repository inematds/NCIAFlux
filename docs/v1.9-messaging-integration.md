# v1.9 - Integracao de Mensagens (WhatsApp, Telegram, Email)

> **Status**: Planejado (apos v1.8 - Times com dados na nuvem)
> **Prerequisito**: Dados sincronizados na nuvem (Vercel KV ou Postgres)
> **Estimativa**: 2-4 semanas de desenvolvimento

---

## Visao Geral

Permitir que usuarios interajam com o NCIAFlux via mensagens externas (WhatsApp, Telegram ou Email), como se fosse o chat interno. A IA processara comandos e executara acoes.

### Casos de Uso

- Usuario manda "crie uma tarefa para amanha: reuniao" via WhatsApp
- Usuario pergunta "quais minhas tarefas de hoje?" via Telegram
- Usuario recebe resumo diario por email
- Usuario ativa modo crise por mensagem quando nao tem acesso ao app

#### Passo 3: Criar Endpoint

```typescript
// apps/web/src/app/api/telegram/webhook/route.ts

import { kv } from '@vercel/kv';

const TELEGRAM_TOKEN = process.env.TELEGRAM_BOT_TOKEN;

export async function POST(request: Request) {
  const body = await request.json();
  const message = body.message;

  if (!message?.text) {
    return Response.json({ ok: true });
  }

  const chatId = message.chat.id;
  const text = message.text;
  const telegramUserId = message.from.id;

  // Comando de vinculacao
  if (text.startsWith('/vincular ')) {
    return handleLinkAccount(chatId, text, telegramUserId);
  }

  // Buscar usuario vinculado
  const userData = await kv.get(`telegram:${telegramUserId}`);

  if (!userData) {
    return sendTelegramMessage(chatId,
      "Voce precisa vincular sua conta primeiro.\n\n" +
      "1. Acesse o app MentesBrilhantes\n" +
      "2. Va em Configuracoes → Integracoes → Telegram\n" +
      "3. Copie o codigo e envie aqui: /vincular CODIGO"
    );
  }

  // Processar mensagem com IA
  const response = await processWithAI(text, userData.userId);

  return sendTelegramMessage(chatId, response);
}

async function handleLinkAccount(chatId: number, text: string, telegramUserId: number) {
  const code = text.replace('/vincular ', '').trim();
  const linkData = await kv.get(`link_code:${code}`);

  if (!linkData) {
    return sendTelegramMessage(chatId, "Codigo invalido ou expirado. Gere um novo no app.");
  }

  // Salvar vinculacao
  await kv.set(`telegram:${telegramUserId}`, {
    userId: linkData.userId,
    linkedAt: new Date().toISOString()
  });

  // Deletar codigo usado
  await kv.del(`link_code:${code}`);

  return sendTelegramMessage(chatId,
    "Vinculado com sucesso! Agora voce pode:\n\n" +
    "- Criar tarefas: 'crie tarefa para amanha'\n" +
    "- Listar tarefas: 'minhas tarefas'\n" +
    "- Brain dump: 'anota: ideia tal'\n" +
    "- Modo crise: 'preciso de ajuda'"
  );
}

async function sendTelegramMessage(chatId: number, text: string) {
  await fetch(`https://api.telegram.org/bot${TELEGRAM_TOKEN}/sendMessage`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ chat_id: chatId, text })
  });

  return Response.json({ ok: true });
}

async function processWithAI(message: string, userId: string) {
  const response = await fetch(`${process.env.NEXT_PUBLIC_APP_URL}/api/chat`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      message,
      userId,
      userName: 'Usuario', // Buscar do KV
      source: 'telegram'
    })
  });

  const data = await response.json();
  return data.message;
}
```

// apps/web/src/app/api/whatsapp/webhook/route.ts

const WHATSAPP_TOKEN = process.env.WHATSAPP_ACCESS_TOKEN;
const WHATSAPP_PHONE_ID = process.env.WHATSAPP_PHONE_NUMBER_ID;
const VERIFY_TOKEN = process.env.WHATSAPP_VERIFY_TOKEN;

// GET - Verificacao do webhook (Meta chama uma vez)
export async function GET(request: Request) {
  const { searchParams } = new URL(request.url);

  const mode = searchParams.get('hub.mode');
  const token = searchParams.get('hub.verify_token');
  const challenge = searchParams.get('hub.challenge');

  if (mode === 'subscribe' && token === VERIFY_TOKEN) {
    return new Response(challenge, { status: 200 });
  }

  return new Response('Forbidden', { status: 403 });
}

// POST - Receber mensagens
export async function POST(request: Request) {
  const body = await request.json();

  const entry = body.entry?.[0];
  const changes = entry?.changes?.[0];
  const value = changes?.value;
  const message = value?.messages?.[0];

  if (!message) {
    return Response.json({ ok: true });
  }

  const from = message.from; // telefone do usuario
  const text = message.text?.body;

  // Verificar vinculacao
  const userData = await kv.get(`whatsapp:${from}`);

  if (!userData) {
    // Verificar se e codigo de vinculacao
    if (text?.startsWith('VINCULAR ')) {
      return handleLinkAccount(from, text);
    }

    return sendWhatsAppMessage(from,
      "Voce precisa vincular sua conta primeiro. " +
      "Acesse o app e va em Configuracoes → WhatsApp"
    );
  }

  // Processar com IA
  const response = await processWithAI(text, userData.userId);
  return sendWhatsAppMessage(from, response);
}

async function sendWhatsAppMessage(to: string, text: string) {
  await fetch(
    `https://graph.facebook.com/v17.0/${WHATSAPP_PHONE_ID}/messages`,
    {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${WHATSAPP_TOKEN}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        messaging_product: 'whatsapp',
        to,
        text: { body: text }
      })
    }
  );

  return Response.json({ ok: true });
}
```

// apps/web/src/app/api/email/inbound/route.ts

export async function POST(request: Request) {
  const formData = await request.formData();

  const from = formData.get('from') as string;
  const subject = formData.get('subject') as string;
  const text = formData.get('text') as string;

  // Extrair email do formato "Nome <email@exemplo.com>"
  const emailMatch = from.match(/<(.+)>/);
  const email = emailMatch ? emailMatch[1] : from;

  // Buscar usuario vinculado
  const userData = await kv.get(`email:${email}`);

  if (!userData) {
    return sendEmail(email, 'Conta nao vinculada',
      'Voce precisa vincular sua conta primeiro no app.'
    );
  }

  // Processar com IA
  const response = await processWithAI(text, userData.userId);

  // Responder por email
  return sendEmail(email, `Re: ${subject}`, response);
}
```

## 4. Endpoint Unificado

Para todas as plataformas usarem a mesma logica:

```typescript
// apps/web/src/app/api/external/chat/route.ts

import { kv } from '@vercel/kv';

interface ExternalChatRequest {
  message: string;
  platform: 'telegram' | 'whatsapp' | 'email';
  platformUserId: string;
  platformUserName?: string;
}

export async function POST(request: Request) {
  // Verificar API key
  const authHeader = request.headers.get('Authorization');
  if (authHeader !== `Bearer ${process.env.EXTERNAL_CHAT_API_KEY}`) {
    return Response.json({ error: 'Unauthorized' }, { status: 401 });
  }

  const body: ExternalChatRequest = await request.json();
  const { message, platform, platformUserId } = body;

  // Buscar vinculacao
  const userData = await kv.get(`${platform}:${platformUserId}`);

  if (!userData) {
    return Response.json({
      success: false,
      message: 'Usuario nao vinculado. Acesse o app para vincular.',
      needsLinking: true
    });
  }

  // Buscar dados do usuario do KV
  const userTasks = await kv.get(`user:${userData.userId}:tasks`) || [];
  const userEvents = await kv.get(`user:${userData.userId}:events`) || [];

  // Chamar chat API existente
  const chatResponse = await fetch(`${process.env.NEXT_PUBLIC_APP_URL}/api/chat`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      message,
      userId: userData.userId,
      userName: userData.userName || 'Usuario',
      context: {
        tasksList: userTasks,
        eventsList: userEvents,
        source: platform
      }
    })
  });

  const data = await chatResponse.json();

  // Salvar acoes no KV (se houver)
  if (data.toolResults) {
    for (const result of data.toolResults) {
      if (result.success) {
        await handleToolResultForKV(result, userData.userId);
      }
    }
  }

  return Response.json({
    success: true,
    message: data.message,
    toolResults: data.toolResults
  });
}
```

## 5. Fluxo de Vinculacao

### Geracao de Codigo

```typescript
// apps/web/src/app/api/integrations/generate-link-code/route.ts

import { kv } from '@vercel/kv';
import { nanoid } from 'nanoid';

export async function POST(request: Request) {
  const { userId, platform } = await request.json();

  // Gerar codigo unico
  const code = nanoid(8).toUpperCase(); // Ex: "ABC12XYZ"

  // Salvar com expiracao de 10 minutos
  await kv.set(`link_code:${code}`, {
    userId,
    platform,
    createdAt: new Date().toISOString()
  }, { ex: 600 }); // 600 segundos = 10 minutos

  return Response.json({ code });
}
```
